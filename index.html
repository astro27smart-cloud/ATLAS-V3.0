async function sync() { const input = document.getElementById('userInput'); const text = input.value.trim(); const status = document.getElementById('status'); if(!text) return; if(activeId === null) { activeId = Date.now(); chats.unshift({ id: activeId, name: "New Sync", history: [] }); } const chat = chats.find(c => c.id === activeId); addUI(text, 'user'); // Your message appears immediately chat.history.push({ role: 'user', text }); input.value = ''; status.innerText = "THINKING..."; // This should change back once he responds try { const res = await fetch('/api/sync', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ prompt: text, history: chat.history, globalMemory: globalMemory }) }); if (!res.ok) throw new Error('Server Down'); const data = await res.json(); // Update the status and handle the memory status.innerText = data.memoryUpdate ? "CORE UPDATED" : "STABLE"; if (data.memoryUpdate) { globalMemory += ` | ${data.memoryUpdate}`; } // SHOW THE ACTUAL RESPONSE addUI(data.text, 'twin'); chat.history.push({ role: 'twin', text: data.text }); save(); } catch (e) { status.innerText = "LINK SEVERED"; addUI("SYSTEM ERROR: I can't reach the backend. Check if your API key is set and the server is running.", 'twin'); } }
